Build Dependency Model (DAG)

Parses the input portfolio and constructs a directed acyclic graph using networkx. Each application becomes a node. Dependencies become directed edges from dependency → dependent app.

import networkx as nx

def build_dependency_graph(portfolio):
dag = nx.DiGraph()
for app in portfolio:
dag.add_node(app['name'], **app)
for dep in app['deps']:
dag.add_edge(dep, app['name'])
return dag

DAG Algorithm

def topological_sort_waves(dag):
waves = []
while dag.nodes():
# Find nodes with no
# incoming edges
wave = [
n for n in dag.nodes()
if dag.in_degree(n) == 0
]
waves.append(wave)
dag.remove_nodes_from(wave)
return waves
